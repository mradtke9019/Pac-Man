#version 330

in vec3 vPosition;
in vec4 vColor;
uniform float time;
uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;

out vec4 color;

void main()                                                              
{
	//gl_Position = vec4(vPosition, 1.0);
	gl_Position = projection * view * model * vec4(vPosition, 1.0);
	color = vColor;
}
